    "class BBoxItem(VBox):\n",
    "    \"\"\"BBox row with select button, coordinates inputs and delete button\"\"\"\n",
    "\n",
    "    def __init__(\n",
    "        self,\n",
    "        bbox_coord: BboxCoordinate,\n",
    "        max_coord_input_values: Optional[BboxCoordinate],\n",
    "        index: int,\n",
    "        options: List[str] = None,\n",
    "        readonly: bool = False\n",
    "    ):\n",
    "        super().__init__()\n",
    "\n",
    "        self.readonly = readonly\n",
    "        self.bbox_coord = bbox_coord\n",
    "        self.index = index\n",
    "        self._max_coord_input_values = max_coord_input_values\n",
    "        self.layout = Layout(display='flex', overflow='hidden')\n",
    "        self.dropdown_classes = self._dropdown_classes(options)\n",
    "        self.btn_select = self._btn_select(index)\n",
    "        self.input_coordinates = self._coordinate_inputs(bbox_coord)\n",
    "\n",
    "        elements = [\n",
    "            self.btn_select,\n",
    "            self.dropdown_classes,\n",
    "            self.input_coordinates,\n",
    "        ]\n",
    "\n",
    "        if not self.readonly:\n",
    "            self.btn_delete = self._btn_delete(index)\n",
    "            elements.append(self.btn_delete)\n",
    "\n",
    "        self.children = [HBox(elements)]\n",
    "\n",
    "    def _btn_delete(self, index: int) -> ActionButton:\n",
    "        return ActionButton(\n",
    "            layout=Layout(width='auto'),\n",
    "            icon=\"trash\",\n",
    "            button_style=\"danger\",\n",
    "            value=index\n",
    "        )\n",
    "\n",
    "    def _dropdown_classes(self, options: Optional[List[str]], value: str = None) -> Dropdown:\n",
    "        return Dropdown(\n",
    "            layout=Layout(width='auto'),\n",
    "            options=options,\n",
    "            value=value,\n",
    "            disabled=self.readonly\n",
    "        )\n",
    "\n",
    "    def _btn_select(self, index: int) -> ActionButton:\n",
    "        return ActionButton(\n",
    "            icon=\"lightbulb-o\",\n",
    "            layout=Layout(width='auto'),\n",
    "            value=index\n",
    "        )\n",
    "\n",
    "    def _coordinate_inputs(self, bbox_coord: BboxCoordinate):\n",
    "        return CoordinateInput(\n",
    "            bbox_coord=bbox_coord,\n",
    "            input_max=self._max_coord_input_values,\n",
    "            disabled=self.readonly\n",
    "        )"

    "#exporti\n",
    "\n",
    "class BBoxList(VBox):\n",
    "    \"\"\"Render the list of bbox items and set the interactions\"\"\"\n",
    "\n",
    "    def __init__(\n",
    "        self,\n",
    "        classes: list,\n",
    "        max_coord_input_values: Optional[BboxCoordinate],\n",
    "        on_coords_changed: Optional[Callable],\n",
    "        on_label_changed: Callable,\n",
    "        on_btn_delete_clicked: Callable,\n",
    "        on_btn_select_clicked: Optional[Callable],\n",
    "        readonly: bool = False\n",
    "    ):\n",
    "        super().__init__()\n",
    "        self._classes = classes\n",
    "        self._max_coord_input_values = max_coord_input_values\n",
    "        self._on_coords_changed = on_coords_changed\n",
    "        self._on_btn_delete_clicked = on_btn_delete_clicked\n",
    "        self._on_label_changed = on_label_changed\n",
    "        self._on_btn_select_clicked = on_btn_select_clicked\n",
    "        self.readonly = readonly\n",
    "\n",
    "    @property\n",
    "    def max_coord_input_values(self) -> Optional[BboxCoordinate]:\n",
    "        return self._max_coord_input_values\n",
    "\n",
    "    @max_coord_input_values.setter\n",
    "    def max_coord_input_values(self, value: BboxCoordinate):\n",
    "        for children in self.children:  # type: ignore\n",
    "            children.input_max = value\n",
    "        self._max_coord_input_values = value\n",
    "\n",
    "    def render_btn_list(self, bbox_coords: List[BboxCoordinate], classes: List[List[str]]):\n",
    "        elements: List[BBoxItem] = []\n",
    "        num_children = len(self.children)  # type: ignore\n",
    "\n",
    "        for index, coord in enumerate(bbox_coords[num_children:], num_children):\n",
    "            bbox_item = BBoxItem(\n",
    "                index=index,\n",
    "                options=self._classes,\n",
    "                bbox_coord=coord,\n",
    "                max_coord_input_values=self._max_coord_input_values,\n",
    "                readonly=self.readonly\n",
    "            )\n",
    "\n",
    "            if not self.readonly:\n",
    "                bbox_item.btn_delete.on_click(self.del_element)\n",
    "            bbox_item.input_coordinates.uuid = index\n",
    "            bbox_item.input_coordinates.coord_changed = self._on_coords_changed\n",
    "            bbox_item.btn_select.on_click(self._on_btn_select_clicked)\n",
    "            if classes and classes[index]:\n",
    "                bbox_item.dropdown_classes.value = classes[index][0] or None\n",
    "            bbox_item.dropdown_classes.observe(\n",
    "                partial(self._on_label_changed, index=index),\n",
    "                names=\"value\",\n",
    "            )\n",
    "\n",
    "            elements.append(bbox_item)\n",
    "\n",
    "        self.children = [*list(self.children), *elements]  # type: ignore\n",
    "\n",
    "    def __getitem__(self, index: int):\n",
    "        return self.children[index]\n",
    "\n",
    "    def clear(self):\n",
    "        self.children = []\n",
    "\n",
    "    def _update_bbox_list_index(self, elements: list, index: int):\n",
    "        for index, element in enumerate(elements[index:], index):\n",
    "            # updates select btn\n",
    "            element.btn_select.value = index\n",
    "            # update label dropdown\n",
    "            dropdown = element.dropdown_classes\n",
    "            dropdown.unobserve_all()\n",
    "            dropdown.observe(\n",
    "                partial(self._on_label_changed, index=index),\n",
    "                names=\"value\"\n",
    "            )\n",
    "            # update inputs\n",
    "            element.children[0].children[2].uuid = index\n",
    "            # updates delete btn\n",
    "            element.btn_delete.value = index\n",
    "\n",
    "    def del_element(self, btn: ActionButton):\n",
    "        index = btn.value\n",
    "        elements = list(self.children)\n",
    "        del elements[index]\n",
    "        self._update_bbox_list_index(elements, index)\n",
    "        self.children = elements\n",
    "        self._on_btn_delete_clicked(index)  # type: ignore"
   ]
